    if (!fbRegex.test(videoLink)) {

        // Logic to fetch video URL and download
        // Note: You'll need a service to actually get the video URL as Facebook does not directly allow it.

        // Send the direct video URL or response to download here
        res.json({ videoUrl: 'Direct video URL goes here' });

    } catch (error) {
        res.status(500).json({ error: 'Failed to fetch video' });
    }
});

// Start server
app.listen(port, () => {
